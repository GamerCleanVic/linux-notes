INSTALAR GIT

INSTALAR GIT NO ARCH LINUX

INSTALAR 

              
sudo pacman -S git && sudo pacman -Syu

SETAR DADOS GITHUB

SETAR EMAIL E USUÁRIO

SETAR EMAIL 

                
git config --global user.email “YourGithubEmail”

SETAR USUÁRIO 

                
git config --global user.name “YourGithubNick”

INICIAR GIT

INICIAR GIT NA PASTA

INICIAR 

                
git init

CRIAR ARQUIVO E ADICIONAR

CRIAR ARQUIVO, ADICIONAR E COMMITAR

CRIE UM ARQUIVO 

                  
index.php

SE PRECISAR VER SE ESTÁ NA BRANCH CORRETA 

                  
git branch

SE PRECISAR BAIXAR AS MUDANÇAS DA BRANCH ONLINE E MESCLAR COM TEUS ARQUIVOS 

                  
git pull

ADICIONAR MUDANÇAS EM TODOS OS ARQUIVOS AO COMMIT 

                  
git add .

COMMITAR AS MUDANÇAS 

                  
git commit -m "QualquerMensagemSobreAsMudanças"

FAZER UPLOAD SINPLES DAS NUDANÇAS 

                  
git push

BRANCH VAZIA

CRIAR BRANCH VAZIA

CRIAR UMA BRANCH VAZIA 

                    
git branch nome_da_branch

BRANCH À PARTIR DE OUTRA

CRIAR NOVA BRANCH À PARTIR DE OUTRA

CRIAR NOVA BRANCH À PARTIR DE OUTRA 

                      
git checkout -b NovaBranchNome BranchAnteriorBase

TROCAR

TROCAR DE BRANCH

TROCAR DE BRANCH 

                        
git checkout NomeDaBrach

MESCLAR

MESCLAR COM OUTRA BRANCH

TROCAR DE BRANCH (P/ NÃO MESCLAR BRANCH ERRADA COM OUTRA) 

                          
git checkout NomeDaBrach

BAIXAR CONTEÚDO DA BRANCH ONLINE E MESCLAR COM A LOCAL EM QUE ESTÁ 

                          
git pull

MESCLAR 

                          
git merge NomeDaBranchAMesclar

VER LOG COMMIT 

                          
git reflog

RESETAR/TROCAR PARA OUTRO COMMIT (SE NECESSÁRIO E AUTORIZADO PELA EQUIPE) 

                          
 git reset --hard NºdoCommit
